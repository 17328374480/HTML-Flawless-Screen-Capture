<script>
(function() {
    // ==================== DOM 元素 ====================
    const startBtn = document.getElementById('startBtn');
    const stopBtn = document.getElementById('stopBtn');

    const bitrateSlider = document.getElementById('bitrateSlider');
    const bitrateDisplay = document.getElementById('bitrateDisplay');
    const codecSelect = document.getElementById('codecSelect');
    const chipContainer = document.getElementById('selectedCodecChip');
    const resWarning = document.getElementById('resWarning');

    const resRadios = document.querySelectorAll('input[name="res"]');
    const fpsRadios = document.querySelectorAll('input[name="fps"]');
    const chromaRadios = document.querySelectorAll('input[name="chroma"]');

    // 音频元素
    const audioEnhanceSelect = document.getElementById('audioEnhance');
    const gainSlider = document.getElementById('gainSlider');
    const gainSpan = document.getElementById('gainValue');
    const audioStatusDiv = document.getElementById('audioStatus');

    gainSlider.addEventListener('input', () => {
        let val = gainSlider.value;
        gainSpan.innerText = (val > 0 ? '+' : '') + val + ' dB';
        if (val > 12) {
            audioStatusDiv.innerHTML = '⚠️ 增益过高，限制器已介入';
        } else {
            audioStatusDiv.innerHTML = '🎧 BBE+限制器 已优化';
        }
    });
    gainSpan.innerText = '+3 dB';

    // 更新编码芯片
    function updateCodecChip() {
        const selected = codecSelect.options[codecSelect.selectedIndex].text;
        chipContainer.innerHTML = `<span class="chip" style="background:#2f4780; border-color:#98b4ff;">🔷 ${selected}</span>`;
    }
    codecSelect.addEventListener('change', updateCodecChip);
    updateCodecChip();

    // 比特率显示
    bitrateSlider.addEventListener('input', (e) => {
        bitrateDisplay.innerText = e.target.value + ' Mbps';
    });
    bitrateDisplay.innerText = bitrateSlider.value + ' Mbps';

    // ==================== 辅助函数 ====================
    function getSelectedResolution() {
        for (let r of resRadios) if (r.checked) {
            const [w, h] = r.value.split('x').map(Number);
            return { width: w, height: h };
        }
        return { width: 3840, height: 2160 };
    }
    
    function getSelectedFPS() {
        for (let f of fpsRadios) if (f.checked) return parseInt(f.value, 10);
        return 60;
    }
    
    function getSelectedChroma() {
        for (let c of chromaRadios) if (c.checked) return c.value;
        return '420';
    }
    
    function getSelectedBitrateMbps() {
        return parseInt(bitrateSlider.value, 10);
    }
    
    function getSelectedCodecKey() {
        return codecSelect.value;
    }

    function buildMimeFromSelection() {
        const codecKey = getSelectedCodecKey();
        const chroma = getSelectedChroma();
        let baseMime = 'video/mp4';
        let codecPart = '';

        switch (codecKey) {
            case 'vp9': baseMime = 'video/webm'; codecPart = 'vp9'; break;
            case 'vp8': baseMime = 'video/webm'; codecPart = 'vp8'; break;
            case 'h264': baseMime = 'video/mp4'; codecPart = 'avc1.640034'; break;
            case 'av1':
                baseMime = 'video/mp4';
                if (chroma === '444') codecPart = 'av01.0.15M.08.0.110.01.01.01.0';
                else if (chroma === '422') codecPart = 'av01.0.15M.08.0.110.01.01.00.0';
                else codecPart = 'av01.0.15M.08';
                break;
            case 'h265': baseMime = 'video/mp4'; codecPart = 'hev1.1.6.L153.B0'; break;
            case 'avc1': baseMime = 'video/mp4'; codecPart = 'avc1.640034'; break;
            case 'vvc': baseMime = 'video/mp4'; codecPart = 'vvc1'; break;
            case 'mp4': baseMime = 'video/mp4'; codecPart = ''; break;
        }
        if (codecPart) return `${baseMime}; codecs="${codecPart}"`;
        return baseMime;
    }

    function getSupportedMimeOrFallback() {
        const desired = buildMimeFromSelection();
        if (MediaRecorder.isTypeSupported(desired)) {
            console.log('✅ 使用选定编码:', desired);
            return desired;
        } else {
            const base = desired.split(';')[0];
            if (MediaRecorder.isTypeSupported(base)) {
                console.warn('⚠️ 特定编码不支持，使用容器:', base);
                alert(`所选编码 ${codecSelect.options[codecSelect.selectedIndex].text} 可能不完全支持，将使用基本容器。`);
                return base;
            } else {
                alert(`错误: 浏览器不支持该编码/容器组合: ${desired}，将尝试video/mp4`);
                return 'video/mp4';
            }
        }
    }

    // ==================== 获取屏幕流 (带系统音频，但video比特率仍由滑块控制) ====================
    async function getCustomStream() {
        const { width, height } = getSelectedResolution();
        const fps = getSelectedFPS();
        
        const constraints = {
            video: {
                width: { ideal: width },
                height: { ideal: height },
                frameRate: { ideal: fps, max: fps }
            },
            audio: {
                echoCancellation: false,
                noiseSuppression: false,
                autoGainControl: false,
                channelCount: 2,
                sampleRate: 48000
            },
            systemAudio: "include"
        };

        try {
            const displayStream = await navigator.mediaDevices.getDisplayMedia(constraints);
            const track = displayStream.getVideoTracks()[0];
            const settings = track.getSettings();
            
            if (settings.width && settings.height) {
                if (settings.width < width || settings.height < height) {
                    resWarning.innerHTML = `⚠️ 实际分辨率: ${settings.width}x${settings.height} (低于选择值，显示器限制)`;
                    resWarning.style.color = '#ffaa66';
                } else {
                    resWarning.innerHTML = `✅ 实际分辨率: ${settings.width}x${settings.height}`;
                    resWarning.style.color = '#66ff99';
                }
            }
            if (settings.frameRate) {
                const fpsInfo = document.createElement('div');
                fpsInfo.style.marginTop = '5px';
                fpsInfo.style.fontSize = '0.9rem';
                if (settings.frameRate < fps) {
                    fpsInfo.innerHTML = `⚠️ 实际帧率: ${Math.round(settings.frameRate)} FPS (低于选择值)`;
                    fpsInfo.style.color = '#ffaa66';
                } else {
                    fpsInfo.innerHTML = `✅ 实际帧率: ${Math.round(settings.frameRate)} FPS`;
                    fpsInfo.style.color = '#66ff99';
                }
                resWarning.appendChild(fpsInfo);
            }
            return displayStream;
        } catch (err) {
            if (err.name === 'NotAllowedError') {
                alert('请授予屏幕共享权限以开始录制');
            } else if (err.name === 'NotFoundError') {
                alert('未找到可录制的屏幕内容');
            } else {
                alert('获取屏幕失败: ' + err.message);
            }
            throw err;
        }
    }

    // ==================== 计时器 ====================
    function formatMinSec(seconds) {
        const m = Math.floor(seconds / 60);
        const s = Math.floor(seconds % 60);
        return `${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
    }
    
    function startTimer() {
        startTime = Date.now();
        timerInterval = setInterval(() => {
            if (!startTime) return;
            const elapsed = Math.floor((Date.now() - startTime)/1000);
            timerDiv.innerHTML = formatMinSec(elapsed) + ' <span>分钟</span>';
        }, 200);
    }
    
    function stopTimer() { 
        if (timerInterval) { clearInterval(timerInterval); timerInterval = null; }
    }
    
    function resetTimer() { timerDiv.innerHTML = '00:00 <span>分钟</span>'; }

    // ==================== 录制状态变量 ====================
    let rawStream = null;
    let audioContext = null;
    let processedStreamForRecord = null;
    let mediaRecorder = null;
    let recordedChunks = [];
    let startTime = null;
    let timerInterval = null;
    let recordingActive = false;
    let stream = null; // 兼容旧引用

    // ==================== 录制回调 ====================
    function onDataAvailable(e) { 
        if (e.data.size > 0) recordedChunks.push(e.data); 
    }

    function cleanupAudioAndStream() {
        if (audioContext && audioContext.state !== 'closed') {
            audioContext.close().catch(e => console.warn('关闭AudioContext失败', e));
        }
        if (rawStream) {
            rawStream.getTracks().forEach(t => t.stop());
            rawStream = null;
        }
        audioContext = null;
        processedStreamForRecord = null;
        stream = null;
    }

    function onStopRecording() {
        stopTimer();
        recordingActive = false;

        cleanupAudioAndStream();

        if (recordedChunks.length === 0) {
            console.warn('没有录制到数据');
            startBtn.disabled = false;
            stopBtn.disabled = true;
            return;
        }
        
        const mime = mediaRecorder ? mediaRecorder.mimeType : 'video/mp4';
        const blob = new Blob(recordedChunks, { type: mime });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;

        const { width, height } = getSelectedResolution();
        const fps = getSelectedFPS();
        const bitrate = getSelectedBitrateMbps();
        const chroma = getSelectedChroma();
        const codecName = codecSelect.options[codecSelect.selectedIndex].text.replace(/[ .()]/g, '');
        const enhanceMode = audioEnhanceSelect.value;
        const enhanceTag = enhanceMode === 'none' ? '' : '_BBE';
        const now = new Date();
        const ts = `${now.getFullYear()}-${now.getMonth()+1}-${now.getDate()}_${now.getHours()}-${now.getMinutes()}`;
        a.download = `${codecName}_${width}x${height}_${fps}fps_${bitrate}M_chroma${chroma}${enhanceTag}_${ts}.mp4`;
        a.click();
        
        setTimeout(() => URL.revokeObjectURL(url), 100);

        recordedChunks = [];
        mediaRecorder = null;
        startBtn.disabled = false;
        stopBtn.disabled = true;
        startTime = null;
    }

    function stopRecording() {
        if (mediaRecorder && mediaRecorder.state !== 'inactive') {
            mediaRecorder.stop();
        } else {
            cleanupAudioAndStream();
            startBtn.disabled = false;
            stopBtn.disabled = true;
            resetTimer();
        }
        stopTimer();
        startTime = null;
    }

    // ==================== 开始录制 (完全尊重原始比特率滑块，只控制视频比特率) ====================
    async function startRecording() {
        try {
            resetTimer();
            recordedChunks = [];
            
            rawStream = await getCustomStream();
            if (!rawStream) return;

            const audioTracks = rawStream.getAudioTracks();
            if (audioTracks.length === 0) {
                audioStatusDiv.innerHTML = '⚠️ 未检测到系统音频，仅录制视频';
                audioStatusDiv.style.color = '#ffaa66';
            } else {
                audioStatusDiv.innerHTML = '🎤 已捕获系统音频';
                audioStatusDiv.style.color = '#66ff99';
            }

            const enhanceMode = audioEnhanceSelect.value;
            const gainDb = parseFloat(gainSlider.value);
            let finalStream = rawStream;

            // BBE增强处理
            if (audioTracks.length > 0 && enhanceMode === 'bbe') {
                try {
                    const audioTrack = audioTracks[0];
                    if (audioTrack.readyState === 'ended' || !audioTrack.enabled) {
                        throw new Error('音频轨道不可用');
                    }

                    audioContext = new (window.AudioContext || window.webkitAudioContext)({
                        latencyHint: 'playback',
                        sampleRate: 48000
                    });
                    
                    if (audioContext.state === 'suspended') {
                        await audioContext.resume();
                    }

                    const sourceStream = new MediaStream([audioTrack]);
                    const sourceNode = audioContext.createMediaStreamSource(sourceStream);

                    const gainNode = audioContext.createGain();
                    gainNode.gain.value = Math.pow(10, gainDb / 20);

                    const compressorNode = audioContext.createDynamicsCompressor();
                    compressorNode.threshold.setValueAtTime(-18, audioContext.currentTime);
                    compressorNode.knee.setValueAtTime(6, audioContext.currentTime);
                    compressorNode.ratio.setValueAtTime(4, audioContext.currentTime);
                    compressorNode.attack.setValueAtTime(0.005, audioContext.currentTime);
                    compressorNode.release.setValueAtTime(0.2, audioContext.currentTime);

                    // BBE滤波器
                    const lowComp = audioContext.createBiquadFilter();
                    lowComp.type = 'peaking';
                    lowComp.frequency.value = 80;
                    lowComp.Q.value = 1.5;
                    lowComp.gain.value = 2;

                    const highExciter = audioContext.createBiquadFilter();
                    highExciter.type = 'highshelf';
                    highExciter.frequency.value = 5000;
                    highExciter.gain.value = 4;

                    const presence = audioContext.createBiquadFilter();
                    presence.type = 'peaking';
                    presence.frequency.value = 3000;
                    presence.Q.value = 2.0;
                    presence.gain.value = 2;

                    sourceNode.connect(lowComp);
                    lowComp.connect(presence);
                    presence.connect(highExciter);
                    highExciter.connect(compressorNode);
                    compressorNode.connect(gainNode);

                    // 硬限制器
                    const limiterNode = audioContext.createDynamicsCompressor();
                    limiterNode.threshold.setValueAtTime(-2, audioContext.currentTime);
                    limiterNode.knee.setValueAtTime(0, audioContext.currentTime);
                    limiterNode.ratio.setValueAtTime(20, audioContext.currentTime);
                    limiterNode.attack.setValueAtTime(0.001, audioContext.currentTime);
                    limiterNode.release.setValueAtTime(0.01, audioContext.currentTime);

                    gainNode.connect(limiterNode);

                    const destNode = audioContext.createMediaStreamDestination();
                    limiterNode.connect(destNode);

                    const processedStream = destNode.stream;
                    const processedAudioTracks = processedStream.getAudioTracks();
                    if (processedAudioTracks.length === 0) {
                        throw new Error('处理后音频轨道为空');
                    }

                    const videoTracks = rawStream.getVideoTracks();
                    finalStream = new MediaStream([
                        ...videoTracks,
                        processedAudioTracks[0]
                    ]);

                    audioStatusDiv.innerHTML = `🎧 BBE+限制器 | 增益 ${gainDb>0?'+':''}${gainDb} dB`;
                    audioStatusDiv.style.color = '#34c759';

                } catch (err) {
                    console.warn('BBE增强处理失败，使用原始音频', err);
                    audioStatusDiv.innerHTML = `⚠️ BBE增强失败，使用原始音频`;
                    audioStatusDiv.style.color = '#ff9500';
                    if (audioContext) {
                        try { await audioContext.close(); } catch (e) {}
                        audioContext = null;
                    }
                }
            } else if (audioTracks.length === 0) {
                audioStatusDiv.innerHTML = '🔇 无音频轨道，仅录制视频';
            } else {
                audioStatusDiv.innerHTML = '🔊 无增强模式，原始音频录制';
            }

            processedStreamForRecord = finalStream;
            stream = finalStream;

            // ---------- 关键修复：只设置视频比特率，不设置音频比特率，完全尊重原始滑块 ----------
            const mimeType = getSupportedMimeOrFallback();
            const bitrateBps = getSelectedBitrateMbps() * 1_000_000;
            
            const options = { 
                mimeType: mimeType, 
                videoBitsPerSecond: bitrateBps   // 音频比特率完全交由浏览器自动管理
            };

            mediaRecorder = new MediaRecorder(finalStream, options);
            mediaRecorder.ondataavailable = onDataAvailable;
            mediaRecorder.onstop = onStopRecording;
            mediaRecorder.onerror = (e) => { 
                console.error('录制错误:', e); 
                stopRecording(); 
            };

            mediaRecorder.start(1000);
            recordingActive = true;
            startTimer();

            startBtn.disabled = true;
            stopBtn.disabled = false;

            rawStream.getVideoTracks()[0].onended = () => {
                if (recordingActive && mediaRecorder?.state !== 'inactive') {
                    stopRecording();
                }
            };

        } catch (err) {
            console.error('启动录制失败:', err);
            cleanupAudioAndStream();
            startBtn.disabled = false;
            stopBtn.disabled = true;
            resetTimer();
            audioStatusDiv.innerHTML = '❌ 录制启动失败';
            audioStatusDiv.style.color = '#ff3b30';
        }
    }

    // 事件绑定
    startBtn.addEventListener('click', startRecording);
    stopBtn.addEventListener('click', stopRecording);

    // 美化radio选中效果
    document.querySelectorAll('.radio-item').forEach(item => {
        const radio = item.querySelector('input[type="radio"]');
        if (radio && radio.checked) item.classList.add('selected');
        radio?.addEventListener('change', function(e) {
            document.querySelectorAll(`input[name="${this.name}"]`).forEach(r => {
                r.closest('.radio-item').classList.remove('selected');
            });
            if (this.checked) this.closest('.radio-item').classList.add('selected');
        });
    });

    window.addEventListener('beforeunload', () => {
        if (mediaRecorder && mediaRecorder.state !== 'inactive') {
            mediaRecorder.stop();
        }
        cleanupAudioAndStream();
    });
})();
</script>
